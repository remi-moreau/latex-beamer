% Define the box style for proofs
\mdfdefinestyle{dembox}{
  backgroundcolor=none,      % transparent background
  linecolor=black,           % line color
  linewidth=3pt,             % line thickness
  leftline=true,             % line only on the left
  topline=false,             % no top line
  bottomline=false,          % no bottom line
  rightline=false,           % no right line
  innertopmargin=0pt,        % inner margins
  innerbottommargin=10pt,
  innerleftmargin=10pt,
  innerrightmargin=10pt
}

% Define the "dem" environment (for proofs)
\newenvironment{dem}[1][Proof]{%
  \begin{mdframed}[style=dembox]
  \noindent\textbf{#1.}\quad
}{%
  \hfill $\square$
  \end{mdframed}
}
